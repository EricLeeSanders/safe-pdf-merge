CREATE DATABASE safe_pdf_merge;

USE safe_pdf_merge;

CREATE TABLE merge (
	row_id INT UNSIGNED NOT NULL AUTO_INCREMENT,
	ip VARCHAR(39),
	merge_time_utc DATETIME,
	file_size BIGINT UNSIGNED,
	num_files INT UNSIGNED,
	PRIMARY KEY (row_id)
)ENGINE=InnoDB;



DELIMITER //

CREATE PROCEDURE sp_I_new_merge
(	IN p_ip VARCHAR(39),
	IN p_file_size BIGINT UNSIGNED,
	IN p_num_files INT UNSIGNED
)
BEGIN
	INSERT INTO merge(ip, merge_time_utc, file_size, num_files) VALUES (p_ip, NOW(), p_file_size, p_num_files);
END//

DELIMITER ;
